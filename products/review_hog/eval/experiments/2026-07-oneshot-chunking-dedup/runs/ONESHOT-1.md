```python
    self, action_id: int, name: Optional[str] = None, description: Optional[str] = None,
    steps: Optional[list[ActionStepInput]] = None,
) -> bool:
    """Replacing steps discards the action's existing trigger conditions."""
    return steps is not None

async def format_dangerous_operation_preview(
    self, action_id: int, name: Optional[str] = None, description: Optional[str] = None,
    steps: Optional[list[ActionStepInput]] = None,
) -> str:
    action = await _fetch_action(self, action_id)
    await self.check_object_access(action, "editor", action="edit")
    return (
        "This will replace all existing trigger conditions for this action:\n"
        f"{format_action_detail(action)}"
    )
```

Renaming/description-only updates (steps is None) can stay ungated since they carry no risk of silently changing what the action matches.
